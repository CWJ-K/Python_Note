## getattr
    return the attribute of objects
    class Test:
        def __init__(self):
            self.name = 'test'
    
    t = Test()
    getattr(t, 'name') # print test

    getattr(t, 'name2') 
    # raise AttributeError: 'Test' object has no attribute 'name2'

    getattr(t, 'name2', 'hi')  # assign a nonexist attribute to t and give a default value, hi
